	concept_name,
	concept_count
FROM (
	SELECT condition_concept_id AS concept_id,
		COUNT(*) AS concept_count
	FROM scratch.dbo.mschuemi_temp
	INNER JOIN CDM_IBM_MDCR_V1062.dbo.condition_occurrence
		ON subject_id = person_id
			AND cohort_start_date = condition_start_date
	INNER JOIN #Codesets
		ON condition_concept_id = concept_id
	WHERE cohort_definition_id = @cohort_id
		AND codeset_id IN (0)
	GROUP BY condition_concept_id
	) concept_counts
INNER JOIN CDM_IBM_MDCR_V1062.dbo.concept
	ON concept_counts.concept_id = concept.concept_id;
